// loop through ALL pairs and names
for (var i = 0; i < directory.length; i++) {
  var colors = directory[i].colors
  var name = directory[i].name
  // Loop through each color pair
  for (var j = 0; j < colors.length; j++) {
    var color = colors[j]
    // Swap colors
    activeDocument.colorReplace(
      color[0], // old color
      color[1], // new color
      0, // tolerance
      false, // anti-alias
      false, // continuous
      false // fill mode
    )
  }

  // Check if the layer has a mask
  var hasMask = false
  try {
    if (
      activeDocument.activeLayer.mask.kind !== LayerKind.NOINTERLOCKCHANNELS
    ) {
      hasMask = true
    }
  } catch (e) {}

  // Export the file
  if (hasMask) {
    activeDocument.exportDocument(
      new File(activeDocument.path + '/primary_' + name),
      ExportType.SAVEFORWEB,
      new ExportOptionsSaveForWeb()
    )

    activeDocument.activeLayer.mask.enabled = true

    activeDocument.exportDocument(
      new File(activeDocument.path + '/secondary_' + name),
      ExportType.SAVEFORWEB,
      new ExportOptionsSaveForWeb()
    )
  } else {
    activeDocument.exportDocument(
      new File(activeDocument.path + '/' + name),
      ExportType.SAVEFORWEB,
      new ExportOptionsSaveForWeb()
    )
  }
}
